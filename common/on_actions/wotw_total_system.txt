on_monthly_pulse_state = {
	on_actions = {
		on_wotc_monthly_pulse_state
	}
}

on_yearly_pulse_state = {
	on_actions = {
		on_wotc_yearly_pulse_state
	}
}

on_monthly_pulse_country = {
	on_actions = {
		on_wotc_monthly_pulse_country
	}
}

on_wotc_monthly_pulse_country = {
	effect = {
		c:WOTW_ALIEN ?= {
			wotw_start_total_war_if_not_at_war = yes
			every_scope_war = {
				add_war_war_support = { target = prev value = 100 } 
			}
		}
	}
}

on_wotc_monthly_pulse_state = {
	effect = {
		if = {
			limit = {
				has_global_variable = wotw_total_war_against_aliens
				AND = {
					c:WOTW_ALIEN ?= controller
					NOT = { c:WOTW_ALIEN ?= owner }
				}
			}
			set_state_owner = c:WOTW_ALIEN
		}
		if = {
			limit = {
				has_global_variable = wotw_total_war_against_aliens
				AND = {
					c:WOTW_ALIEN ?= owner
					NOT = { c:WOTW_ALIEN ?= controller }
				}
			}
			set_state_owner = controller
		}
	}
}

on_wotc_yearly_pulse_state = {
	trigger = {
		NOT = {
			has_global_variable = wotw_aliens_whacked
		}
		OR = {
			AND = {
				has_game_rule = wotw_aliens_normal_delay
				year >= 1846
			}
			AND = {
				has_game_rule = wotw_aliens_short_delay
				year >= 1841
			}
			has_game_rule = wotw_aliens_no_delay
		}
	}

	random_events = {
		1 = wotw_crash_sites.1
		4 = 0
	}
}
